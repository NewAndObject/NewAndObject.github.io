移动和动画帧实现代码学习记录

悯 2026-01-30 12:01:00 Godot
#总结移动实现思路
#1.创建2D角色物理体CharacterBody2D节点，并在该节点写创建精灵Sprite2D节点和碰撞节点CollisionSpape2D节点
#2.项目设置里创建上下左右映射
#2.写gd代码，逻辑是使用Input类的方法获取键盘或手柄的输入的上下左右，将输入化为xy坐标赋予2D角色物理体节点并调用move_and_slide()方法	
	
#总结动画帧实现思路
#1.在CharacterBody2D-2D角色身体里创建一个AnimationPlayer节点并添加多帧图片。然后找到下面的动画添加动画
#动画轨道绑定节点创建方法有两种：在底部动画里选中+，选中轨道属性，选中想要的精灵节点。另一个方法是左侧菜单栏选中需要的精灵节点，找到动画Animation下的Frame属性按钥匙后就会自动添加轨道属性
#2.在新建的AnimationPlayer里的创建动画和静态和行走的每一帧，有默认的reset、静止的上下侧、走动的上下侧总共7个动画，还有一点，创建frame轨道动画关键帧，我们只创建两个，起始和结束，并将这个轨道设置为连续类型，这样非常节省素材，还有一点，还需要在下面的动画设置循环播放
#3.写4个方法用于判断方向和状态并更新方向和状态

class_name Player extends CharacterBody2D
# 加速度变量
@export var move_seep : float =100.0
# 动画类变量，快捷方法：可以按Ctrl+鼠标左键选中的节点托至代码里
@onready var animation_player : AnimationPlayer = $AnimationPlayer
# 精灵类变量，快捷方法如上
@onready var sprite_2d : Sprite2D = $Sprite2D
#状态参数，是静止的idle还是行走的walk，默认静止
var state : String ="idle"
#初始方向变量，默认向下
var cardinal_direction : Vector2 = Vector2.DOWN
#初始位置变量，默认（0，0）
var direction : Vector2 = Vector2.ZERO

# 内置函数，只执行一次，优先级弱于_process()和_physical_process()
func _ready():
	pass # Replace with function body.
#get_input()的三种实现思路，Michael使用了第三种方法
			  
func get_inputOne():
	#输入方向获取，调用Input的get_vector方法，就可以获取你的上下左右改变的坐标值，是get_inputTwo和get_input的最精简写法
	direction=Input.get_vector("left","right","up","down")
	velocity=direction*move_seep
	pass
func get_inputTwo():
	
	#Input.get_axis()方法是Input.get_action_strength(正数)-Input.get_action_strength(坐标)的精简写法
	direction.x=Input.get_axis("left","right")
	direction.y=Input.get_axis("up","down")
	velocity=direction*move_seep
	pass
func get_input():
	#获取xy轴坐标，获取方法是：正参数减负参数。在godot里，xy坐标轴中，x轴正坐标在右边的坐标轴上，y轴的正坐标在下面的坐标轴上
	direction.x=Input.get_action_strength("right")-Input.get_action_strength("left")
	direction.y=Input.get_action_strength("down")-Input.get_action_strength("up")
	#按一下键盘只是移动1个像素，为了移动更明显，乘于加速度参数move_speed
	velocity=direction*move_seep
	pass
# 内置周期函数，1/60秒执行，优先级弱于_physical_process()内置函数
func _process(delta):
	#print("velocity:",velocity)
	get_input()
	#在状态和方向二者某一个发生变化时，更新动画状态
	if SetState()==true||SetDirection()==true:
		UpdateAnimation()
	pass
# 内置函数，1/60秒执行
func _physics_process(delta):
	#内置函数，更新移动坐标
	move_and_slide()
	pass
#设置方向方法，返回bool类型
func SetDirection() -> bool:
	#创建局部变量new_direction获取初始方向
	var new_direction:Vector2=cardinal_direction
	#判断获取了最新初始方向的局部变量是否有变化，没有返回false
	if direction==Vector2.ZERO:
		return false
	#判断，在y轴没有变化时，新的方向是否往左变化（即全局变量direction的x属性是否小于0），变化了方向为-1（也可以写成Vector2.LEFT），否则相反
	if direction.y==0:
		new_direction=Vector2.LEFT if direction.x<0 else Vector2.RIGHT
	#判断参考上
	if direction.x==0:
		new_direction=Vector2.UP if direction.y<0 else Vector2.DOWN
	#判断，新方向是否和初始方向相同，相同就返回false。这里可能会有疑惑：新方向不是在上面获取了初始方向的值嘛，肯定相同啊，我的思考结果是：上面两段代码可能改变了new_direction的新值
	if new_direction==cardinal_direction:
		return false
	#将新方向赋予初始方向，初始方向发生变化
	cardinal_direction=new_direction
	#左右翻转，因为左右只有side侧边动画一个，要实现左右动画可以把侧边动画沿y轴反转，利用精灵节点的scale属性
	sprite_2d.scale.x=-1 if cardinal_direction==Vector2.LEFT else 1
	return true
#设置行走静止状态方法，返回值bool
func SetState() -> bool:
	#创建新状态局部变量，如果全局变量方向是（0，0）即Vector2.ZERO，就默认是静止idle，反之是walk行走
	var new_state :String="idle" if direction==Vector2.ZERO else "walk"
	#判断，新状态和全局变量：状态相同，就返回false
	if new_state==state:
		return false
	#把新状态赋值给全局变量：状态
	state=new_state
	return true
#更新动画方法
func UpdateAnimation() -> void:
	#调用动画节点的播放方法，方法参数为你创建的动画名字：idle_up、walk_down等。
	#因为动画每次按键动画有方向和状态变化，所以需要这两个参数时时变化，state状态参数变化参考SetState()方法，方向参数变化参考SetDirection()方法和Aimdirection()方法
	animation_player.play(state + "_"+AimDirection())
	pass
#动画方向获取方法，返回String
func AimDirection()->String:
	#获取最新的初始方向，如果方向向下，返回“down”，如果方向向上，返回“up”，否则返回侧面“side”
	if cardinal_direction==Vector2.DOWN:
		return "down"
	elif cardinal_direction==Vector2.UP:
		return "up"
	else:
		return "side"
