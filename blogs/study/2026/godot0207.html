## 设置方向方法，返回bool类型
func set_direction(_new_direction:Vector2) -> bool:
	direction=_new_direction
	## 判断获取了最新初始位置的局部变量是否有变化，没有返回false
	if direction==Vector2.ZERO:
		return false
	## 获取方向id索引，解决动画人物太空步问题
	var direction_id:int=int(round(
		(direction+cardinal_direction*0.1).angle()
		/TAU*DIR_4.size()
	))
	## 根据方向id索引获取新方向Vector2属性
	var new_direction=DIR_4[direction_id]
	
	## 判断，新方向是否和初始方向相同，相同就返回false。这里可能会有疑惑：新方向不是在上面获取了初始方向的值嘛，肯定相同啊，我的思考结果是：新方向在代码58到62这里变化了
	if new_direction==cardinal_direction:
		return false
	## 将新方向赋予初始方向，初始方向发生变化
	cardinal_direction=new_direction
	direction_changed.emit(new_direction)
	## 左右翻转，因为左右只有side侧边动画一个，要实现左右动画可以把侧边动画沿y轴反转，利用精灵节点的scale属性
	sprite.scale.x=-1 if cardinal_direction==Vector2.LEFT else 1
	return true
## 更新动画方法
func update_animation(state:String) -> void:
	#调用动画节点的播放方法，方法参数为你创建的动画名字：idle_up、walk_down等。
	#因为动画每次按键动画有方向和状态变化，所以需要这两个参数时时变化，state状态参数变化参考SetState()方法，方向参数变化参考SetDirection()方法和Aimdirection()方法
	animation_player.play(state + "_"+aim_direction())
	pass
## 动画方向获取方法，返回String
func aim_direction()->String:
	#获取最新的初始方向，如果方向向下，返回“down”，如果方向向上，返回“up”，否则返回侧面“side”
	if cardinal_direction==Vector2.DOWN:
		return "down"
	elif cardinal_direction==Vector2.UP:
		return "up"
	else:
		return "side"

class_name EnemeyStateMachine extends Node

## 状态集合变量
var states:Array[EnemyState]
## 临时状态缓存变量，记录上一个状态情况
var prev_state:EnemyState
## 当前状态变量
var current_state:EnemyState
# Called when the node enters the scene tree for the first time.
func _ready() -> void:
	## 防止第一次加载时触发代码
	process_mode=Node.PROCESS_MODE_DISABLED
	pass # Replace with function body.


## 过程内置函数
func _process(delta: float) -> void:
	## 改变状态，参数为当前状态变量
	change_state(current_state.process(delta))
	pass
## 物理过程内置函数
func _physics_process(delta: float) -> void:
	## 改变状态，参数为当前状态变量
	change_state(current_state.physics(delta))
	pass
## 初始化方法
func initialize(_enemy:Enemy)->void:
	states=[]
	## 获取该节点下的子节点
	for c in get_children():
		## 判断该节点是否是敌人状态节点类型，是就添加到状态集合变量中
		if c is EnemyState:
			states.append(c)
	## 获取状态集合
	for s in states:
		## 将每个状态的敌人变量、状态管理器变量赋值，并调用它的初始化方法
		s.enemy=_enemy
		s.state_machine=self
		s.init()
	## 判断状态集合数量是否大于0，大于就调用改变状态方法，参数为状态集合的第一个数据，并将节点开启为可继承
	if states.size()>0:
		change_state(states[0])
		process_mode=Node.PROCESS_MODE_INHERIT
		
		
	pass
## 改变状态方法
func change_state(new_state:EnemyState)->void:
	## 判断是否新状态参数是否为空或等于当前状态，是就返回不执行如何操作
	if new_state==null || new_state==current_state:
		return
	## 判断当前状态是否为空，不是就退出
	if current_state:
		current_state.exit()
	## 之前的状态等于当前状态
	prev_state=current_state
	## 当前状态等于新状态
	current_state=new_state
	## 当前状态调用进入方法
	current_state.enter()
