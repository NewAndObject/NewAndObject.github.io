class_name  State_Attack extends State
# 行走、闲置节点只读变量，快捷创建方法：左边节点菜单栏，按ctrl+鼠标左键拖过来
@onready var walk: State_Walk = $"../Walk"
@onready var idle: State_Idle = $"../Idle"
# 创建玩家动画只读变量
@onready var animation_player: AnimationPlayer = $"../../AnimationPlayer"
# 创建攻击动画只读变量
@onready var attack_anim: AnimationPlayer = $"../../Sprite2D/AttackEffectSprite/AnimationPlayer"
# 创建导出攻击音效，用于存放具体音效
@export var attack_sound:AudioStream
# 攻击行走间的缓存空间变量（range三个参数，第一个是最小值，第二个为最大值，第三个是每次增值大小），具体表现为攻击后会行走一段距离，不会立马停顿
@export_range(1,20,0.5) var decelerate_speed:float=5
# 只读音效变量，一个Node下的AudioStreamPlayer2D节点，该节点不存放音效，用于中转存放音效，比如你需要存放攻击就存放攻击，要存放下雨声就存放下雨声
@onready var audio :AudioStreamPlayer2D = $"../../Audio/AudioStreamPlayer2D"

# 是否攻击，默认false
var attacking:bool=false

func _ready() -> void:
	pass 

# 该方法控制：当玩家进入该状态时发生
func Enter()->void:
	## 玩家更新为攻击动画
	player.UpdateAnimation("attack")
	## 攻击动画更新
	attack_anim.play("attack_"+player.AimDirection())
	## 完成动画时，退出将是否攻击变量修改为：false
	animation_player.animation_finished.connect(EndAttack)
	## 音效播放，音效节点获取具体存放音效的变量（导出变量，导出了就可以在检查器的属性里查看，也就可以添加具体音效文件了）
	audio.stream=attack_sound
	## 音效声音大小属性，用随机函数randf_range(最小值，最大值)
	audio.pitch_scale=randf_range(0.9,1.1)
	## 调用音效节点播放方法
	audio.play()
	## 是否攻击设置为true
	attacking=true
	pass
	
# 该方法控制：当玩家退出该状态时发生
func Exit()->void:
	## 玩家动画节点完成播放后，调用动画播放完成方法结束攻击动画，并调用结束动画方法（创建这个结束动画方法目前只有将attacking值改为false的业务，如果后期有其他业务，也可以添加）
	animation_player.animation_finished.disconnect(EndAttack)
	pass

# 该方法控制：当_process发生期间时状态进行的操作
func Process(_delta:float)->State:
	## 玩家坐标不能直接设置为0，0，因为这样会导致行走按攻击会有明显停顿，所以让行走在移动一点点，我打个比方：比如我行走中并攻击了，这期间移动了0，2，那我们播放攻击动画的过程中，还要移动0.01位置，以达到连续性
	player.velocity-=player.velocity*decelerate_speed*_delta
	## 判断是否按了攻击，没有（并且玩家初始位置没有发生变化（就是玩家没按上下左右移动），就返回闲置状态，反之闲置状态），有返回空
	if attacking==false:
		if player.direction==Vector2.ZERO:
			return idle
		else:
			return walk
	
	return null
