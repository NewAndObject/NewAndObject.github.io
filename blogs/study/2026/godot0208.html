class_name Enemy extends CharacterBody2D
## 方向变化信号
signal direction_changed(new_direction:Vector2)
## 敌人受到伤害信号
signal enemy_damaged()
## 4个方向的常量,顺序分别是：右，下，左，上
const DIR_4=[Vector2.RIGHT,Vector2.DOWN,Vector2.LEFT,Vector2.UP]
## 敌人血量
@export var hp:int=3
## 之前方向变量，默认向下,存放之前的方向向量坐标
var cardinal_direction : Vector2 = Vector2.DOWN
## 初始位置变量，默认（0，0）
var direction : Vector2 = Vector2.ZERO
## 玩家变量，还没用上
var player:Player
## 是否会受到伤害
var invulnerable:bool=false
## 动画变量
@onready var animation_player: AnimationPlayer = $AnimationPlayer
## 精灵变量
@onready var sprite: Sprite2D = $Sprite2D
## 状态管理器变量-只读
@onready var state_machine:EnemeyStateMachine=$EnemyStateManchine
## 获取碰撞盒子节点场景
@onready var hit_box: HitBox = $HitBox
## 销毁信号
signal enemy_destroyed()

# Called when the node enters the scene tree for the first time.
func _ready() -> void:
	## 状态管理器初始化加载方法调用
	state_machine.initialize(self)
	player=PlayerManager.player
	hit_box.Damage.connect(take_damage)

func _process(delta: float) -> void:
	pass

func _physics_process(delta: float) -> void:
	move_and_slide()
	
	pass

## 设置方向方法，返回bool类型
func set_direction(_new_direction:Vector2) -> bool:
	direction=_new_direction
	## 判断获取了最新初始位置的局部变量是否有变化，没有返回false
	if direction==Vector2.ZERO:
		return false
	## 获取方向id索引，解决动画人物太空步问题
	var direction_id:int=int(round(
		(direction+cardinal_direction*0.1).angle()
		/TAU*DIR_4.size()
	))
	## 根据方向id索引获取新方向Vector2属性
	var new_direction=DIR_4[direction_id]
	
	## 判断，新方向是否和初始方向相同，相同就返回false。这里可能会有疑惑：新方向不是在上面获取了初始方向的值嘛，肯定相同啊，我的思考结果是：新方向在代码58到62这里变化了
	if new_direction==cardinal_direction:
		return false
	## 将新方向赋予初始方向，初始方向发生变化
	cardinal_direction=new_direction
	direction_changed.emit(new_direction)
	## 左右翻转，因为左右只有side侧边动画一个，要实现左右动画可以把侧边动画沿y轴反转，利用精灵节点的scale属性
	sprite.scale.x=-1 if cardinal_direction==Vector2.LEFT else 1
	return true
## 更新动画方法
func update_animation(state:String) -> void:
	#调用动画节点的播放方法，方法参数为你创建的动画名字：idle_up、walk_down等。
	#因为动画每次按键动画有方向和状态变化，所以需要这两个参数时时变化，state状态参数变化参考SetState()方法，方向参数变化参考SetDirection()方法和Aimdirection()方法
	animation_player.play(state + "_"+aim_direction())
	pass
## 动画方向获取方法，返回String
func aim_direction()->String:
	#获取最新的初始方向，如果方向向下，返回“down”，如果方向向上，返回“up”，否则返回侧面“side”
	if cardinal_direction==Vector2.DOWN:
		return "down"
	elif cardinal_direction==Vector2.UP:
		return "up"
	else:
		return "side"
## 受到伤害方法,参数为受到伤害的值
func take_damage(damage:HurtBox)->void:
	# 无敌状态，不扣血
	if invulnerable==true:
		return
	hp-=damage.damage
	if hp<=0:
		enemy_destroyed.emit()
	else:
		enemy_damaged.emit()
	
	pass

var _direction:Vector2

	pass # Replace with function body.

func _process(delta: float) -> void:
	pass
## 初始化
func init()->void:
	# 连接敌人受到伤害信号
	enemy.enemy_damaged.connect(_on_enemy_damaged)
	pass

## 该方法控制：当玩家进入该状态时发生
func enter()->void:
	# 当进入被攻击动画时，就会处于无敌状态
	enemy.invulnerable=true
	animation_finished=false
	# 位置为玩家位置
	_direction=enemy.global_position.direction_to(enemy.player.global_position)
	enemy.set_direction(_direction)
	enemy.velocity=_direction*-knockback_speed
	enemy.update_animation(anmi_name)
	enemy.animation_player.animation_finished.connect(_on_animation_finished)
	pass
	
## 该方法控制：当玩家退出该状态时发生
func exit()->void:
	enemy.invulnerable=false
	enemy.animation_player.animation_finished.disconnect(_on_animation_finished)
	pass

## 该方法控制：当_process发生期间时状态进行的操作
func process(_delta:float)->EnemyState:
	if animation_finished==true:
		return next_state
	enemy.velocity-=enemy.velocity*decelerate_speed*_delta
	return null
	
## 该方法控制：当_physics_process发生期间执行的操作
func physics(_delta:float)->EnemyState:
	
	return null
## 敌人受到伤害改变状态方法
func _on_enemy_damaged()->void:
	state_machine.change_state(self)
	pass
## 动画结束方法
func _on_animation_finished(a:String)->void:
	animation_finished=true
	pass

class_name HurtBox extends Area2D
## 默认伤害值
@export var damage:int=1

# Called when the node enters the scene tree for the first time.
func _ready() -> void:
	## 当接收的 area 进入此区域时发出。需要 monitoring 被设置为 true。将信号连接到。
	area_entered.connect(AreaEntered)
	pass # Replace with function body.


# Called every frame. 'delta' is the elapsed time since the previous frame.
func _process(delta: float) -> void:
	pass
## 区域进入方法，参数为区域节点参数Area2D
func AreaEntered(a:Area2D)->void:
	## 判断传入的是否为HitBox类型，是就调用HitlerBox类的方法TakeDamage()
	if a is HitBox:
		a.TakeDamage(self)
	pass

  • 说明
1. 新的动画帧，可以用属性轨道的（Sprite）的modulate属性改变精灵的颜色，达到被攻击变化
2. 创建的音频节点不需要赋值，直接在动画帧里添加，直接调用并将音频拖到轨道上，另外音频节点（AudioStreamPlayer2D）有个Max Polyhony属性可以设置最大音频数量
3. 轨道属性的（Sprite）可以创建visible属性轨道，用于调整攻击动画的消失
4. 轨道属性的（Sprite）可以创建position属性轨道，用于调整敌人动画的移动，这样敌人就有被攻击后移动的变化
