  • 玩家场景下的攻击特效动画节点代码
class_name PlayerInteractionHost extends Node2D

@onready var player: Player = $".."

# Called when the node enters the scene tree for the first time.
func _ready() -> void:
	player.ChangeDirection.connect(UpdateDirection)
	pass # Replace with function body.


# 改变攻击范围方向方法
func UpdateDirection(new_direction:Vector2)->void:
	match new_direction:
		Vector2.DOWN:
			rotation_degrees=0
		Vector2.UP:
			rotation_degrees=180
		Vector2.LEFT:
			rotation_degrees=90
		Vector2.RIGHT:
			rotation_degrees=-90
		_:
			rotation_degrees=0
	pass

  1. 其他说明
  • HitBox和HurtBox场景的可碰撞节点不再使用CollisionShape2D节点，而是使用Area2D节点，目的是：使用Monitoring属性和Monitorable属性，用于检测攻击节点和被攻击节点，HitBox场景：Monitoring设置为false，Monitorable设置为true，HurtBox场景：Monitoring设置为true，Monitorable设置为false
  • Plant场景还需要使用StaticBody2D节点，作用于使被攻击的物体不被玩家移动
  • HitBox场景：Layer 9，Mask
  • HurtBox场景：Layer 2，Mask
  • Plant场景：Layer 5，Mask
  • Player场景：Layer 1，Mask 5
